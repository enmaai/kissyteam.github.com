function parser(inputVal, data) {
    var ret = [],
        count = 0;

    if (!inputVal) {
        return data;
    }

    S.each(data, function (d) {
        if (d.indexOf(inputVal) != -1) {
            ret.push(d);
        }
        count++;
    });

    return ret;
}
